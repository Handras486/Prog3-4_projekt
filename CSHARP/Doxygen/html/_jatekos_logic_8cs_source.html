116  join csapat in this.CsapatRepository.GetTableElements().AsEnumerable() on jatekos.Csapatnev equals csapat.Csapatnev
117  join liga in this.LigaRepository.GetTableElements().AsEnumerable() on csapat.Liga_nev equals liga.Liga_nev
118  where jatekos.Felhasznalonev == felhasznalonev
119  select new { jatekos.Felhasznalonev, csapat.Csapatnev, liga.Liga_nev };
120 
121  var query2 = from jatekos in this.JatekosRepository.GetTableElements().AsEnumerable()
122  where jatekos.Csapatnev == this.JatekosRepository.GetTableElement(felhasznalonev).Csapatnev
123  group jatekos by jatekos.Csapatnev into g
124  orderby g.Count()
125  select g.Count();

127  foreach (var q in query1)
